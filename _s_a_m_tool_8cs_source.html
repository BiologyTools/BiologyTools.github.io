145 static List<PointD> GetEdgePolygonFromMask(Bitmap maskBitmap)
146 {
147 List<PointD> edgePolygon = new List<PointD>();
148
149 // Lock the bitmap data to access pixel values
150 BitmapData data = maskBitmap.LockBits(new AForge.Rectangle(0, 0, maskBitmap.Width, maskBitmap.Height),
151 ImageLockMode.ReadOnly, PixelFormat.Format32bppArgb);
152
153 unsafe
154 {
155 byte* ptr = (byte*)data.Scan0;
156
157 for (int y = 1; y < maskBitmap.Height - 1; y++)
158 {
159 for (int x = 1; x < maskBitmap.Width - 1; x++)
160 {
161 int index = y * data.Stride + x * 4;
162
163 // Check if the pixel is non-zero (white)
164 if (ptr[index] > 0)
165 {
166 // Check if at least one neighboring pixel is zero (black)
167 if (ptr[index - data.Stride] == 0 || ptr[index + data.Stride] == 0 || ptr[index - 4] == 0 || ptr[index + 4] == 0)
168 {
169 edgePolygon.Add(new PointD(x, y));
170 }
171 }
172 }
173 }
174 }
175
176 // Unlock the bitmap data
177 maskBitmap.UnlockBits(data);
178
179 return OrderPolygon(edgePolygon);
180 }
181 static List<PointD> OrderPolygon(List<PointD> polygon)
182 {
183 // Calculate the centroid of the polygon
184 PointD centroid = new PointD((int)polygon.Average(p => p.X), (int)polygon.Average(p => p.Y));
185
186 // Sort the points based on polar angle
187 List<PointD> orderedPolygon = polygon.OrderBy(p => Math.Atan2(p.Y - centroid.Y, p.X - centroid.X)).ToList();
188
189 return orderedPolygon;
190 }

260 private void PictureBox_ButtonReleaseEvent(object o, ButtonReleaseEventArgs e)

262 if (view.Images.Count < 1)

269 {
270 OpType type = this.addMaskMenu.Active == true ? OpType.ADD : OpType.REMOVE;
271 Promotion promtt = new PointPromotion(type);
272 (promtt as PointPromotion).X = (int)SelectedROI.PointsImage[0].X;
273 (promtt as PointPromotion).Y = (int)SelectedROI.PointsImage[0].Y;

276 this.mPromotionList.Add(ptn);

279 this.mPromotionList.Clear();

281
282 if (Tools.currentTool.type == Tools.Tool.Type.rect && e.Event.State.HasFlag(ModifierType.Button1Mask))
283 {
284 BoxPromotion promt = new BoxPromotion();

294 (promt as BoxPromotion).mLeftUp.X = (int)SelectedROI.PointsImage[3].X;

349 void ShowMask(float[] mask, int width, int height)
350 {
351 Bitmap bp = new Bitmap(width, height, PixelFormat.Format32bppArgb);
352 byte[] pixelData = new byte[width * height * 4];
353 Array.Clear(pixelData, 0, pixelData.Length);
354 for (int y = 0; y < height; y++)
355 {
356 for (int x = 0; x < width; x++)
357 {
358 int ind = y * width + x;
359 if (mask[ind] > thresholdBar.Value)
360 {
361 pixelData[4 * ind] = 255;// Blue
362 pixelData[4 * ind + 1] = 255;// Green
363 pixelData[4 * ind + 2] = 255;// Red
364 pixelData[4 * ind + 3] = 255;// Alpha
365 }
366 else
367 pixelData[4 * ind + 3] = 0;
368 }
369 }
370 Pixbuf pf = new Pixbuf(pixelData, true, 8, width, height, width*4);
371 this.SelectedROI.mask = pf;
372 if (autoSaveMenu.Active)
373 {
374 //Saving Tiff can cause a crash on Windows so we check the platform.
375 if (Environment.OSVersion.Platform == PlatformID.Win32NT)
376 {
377 pf.Save(autoSavePath + "/" + textBox.Text + id + ".png", "png");
378 }
379 else
380 {
381 pf.Save(autoSavePath + "/" + textBox.Text + id + ".tiff", "tiff");
382 }
383 id++;
384 }
