99 double[,] BuildPointGrid(int n_per_side)
100 {
101 double offset = 1.0 / (2 * n_per_side);
102 double[] points_one_side = Enumerable.Range(0, n_per_side)
103 .Select(i => offset + i * (1.0 - 2 * offset) / (n_per_side - 1))
104 .ToArray();
105
106 double[,] points = new double[n_per_side * n_per_side, 2];
107
108 for (int i = 0; i < n_per_side; i++)
109 {
110 for (int j = 0; j < n_per_side; j++)
111 {
112 int index = i * n_per_side + j;
113 points[index, 0] = points_one_side[i];
114 points[index, 1] = points_one_side[j];
115 }
116 }
117
118 return points;
119 }

126 void generateCropBoxes(int orgWid, int orgHei, int n_layers, float overlap_ratio,
127 ref List<List<int>> crop_boxes, ref List<int> layer_idxs)
128 {
129 int im_h = orgHei;
130 int im_w = orgWid;
131
132 int short_side = Math.Min(im_h, im_w);
133 //Original image
134 crop_boxes.Add(new List<int> { 0, 0, im_w, im_h });
135 layer_idxs.Add(0);
136
137 for (int i_layer = 0; i_layer < n_layers; i_layer++)
138 {
139 int n_crops_per_side = (int)Math.Pow(2, i_layer + 1);
140 int overlap = (int)(overlap_ratio * short_side * (2 / n_crops_per_side));
141
142 int crop_w = crop_len(im_w, n_crops_per_side, overlap);
143 int crop_h = crop_len(im_h, n_crops_per_side, overlap);
144
145 List<int> crop_box_x0 = new List<int>();
146 List<int> crop_box_y0 = new List<int>();
147
148 for (int i = 0; i < n_crops_per_side; i++)
149 {
150 crop_box_x0.Add((crop_w - overlap) * i);
151 crop_box_y0.Add((crop_h - overlap) * i);
152 }
153
154 foreach (int x0 in crop_box_x0)
155 {
156 foreach (int y0 in crop_box_y0)
157 {
158 List<int> box = new List<int>
159 {
160 x0,
161 y0,
162 Math.Min(x0 + crop_w, im_w),
163 Math.Min(y0 + crop_h, im_h)
164 };
165
166 crop_boxes.Add(box);
167 layer_idxs.Add(i_layer + 1);
168 }
169 }
170
171 }

173 int crop_len(int orig_len, int n_crops, int overlap)
174 {
175 return (int)(Math.Ceiling((double)(overlap * (n_crops - 1) + orig_len) / n_crops));

177 IEnumerable<List<object>> BatchIterator(int batch_size, params object[] args)
178 {
179
180 int n_batches = ((Array)args[0]).Length / batch_size + (((Array)args[0]).Length % batch_size != 0 ? 1 : 0);
181 for (int b = 0; b < n_batches; b++)
182 {
183 List<object> batch = new List<object>();
184 foreach (object arg in args)
185 {
186 Array arr = (Array)arg;
187 int start = b * batch_size;
188 int end = (b + 1) * batch_size;
189 if (end > arr.Length)
190 end = arr.Length;
191
192 Array slice = Array.CreateInstance(arr.GetType().GetElementType(), end - start);
193 Array.Copy(arr, start, slice, 0, end - start);
194 batch.Add(slice);
195 }
196
197 yield return batch;
198 }

231 MaskData _process_crop(List<int> crop_box, int crop_layer_idx)
