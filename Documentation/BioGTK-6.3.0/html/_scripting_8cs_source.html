234 private static async void RunScript()
235 {
236 Script rn = scripts[scriptName];
237 rn.ex = null;
238 if (rn.type == ScriptType.imagej)
239 {
240 try
241 {
242 rn.done = false;
243 if (App.UseFiji)
244 await Fiji.RunStringFiji(ImageView.SelectedImage,rn.scriptString, "", false);
245 else
246 ImageJ.RunString(rn.scriptString, "", false);
247 rn.done = true;
248 }
249 catch (Exception e)
250 {
251 rn.ex = e;
252 }
253 }
254 else
255 {
256 try
257 {
258 rn.done = false;
259 rn.script = CSScript.Evaluator.LoadCode(rn.scriptString);
260 rn.obj = rn.script.Load();
261 rn.output = rn.obj.ToString();
262 rn.done = true;
263
264 }
265 catch (Exception e)
266 {
267 rn.ex = e;
268 error = e.Message + Environment.NewLine + e.StackTrace;
269 }
270 }
