103 private void Buffer_Changed(object sender, EventArgs e)
104 {
105 if (consoleBox.Buffer.CharCount == 0)
106 return;
107 if(skip)
108 {
109 skip = false;
110 return;
111 }
112 pred = pred.Replace("\t", "").Replace("\r","").Replace("\n","");
113 if (pred.Contains(' '))
114 pred = consoleBox.Buffer.Text.Remove(0,consoleBox.Buffer.Text.LastIndexOf(' '));
115 else
116 pred = consoleBox.Buffer.Text.Replace("\t", "").Replace("\r", "").Replace("\n", "");
117 bool ends = false;
118 char ch = ' ';
119 int i = 0;
120 int ind = -1;
121 foreach (Char cs in pred)
122 {
123 ends = false;
124 foreach (Char c in chs)
125 {
126 if (cs == c)
127 {
128 ends = true;
129 ch = c;
130 }
131 }
132 if (!ends)
133 ind = i;
134 i++;
135 }
136 if (ind!=-1)
137 pred = pred.Remove(0, ind+1);
138
139 preds.Clear();
140 i = 0;
141 ind = -1;
142 foreach(var cs in Fiji.Macro.Commands)
143 {
144 if (cs.Value.Name.StartsWith(pred))
145 preds.Add(cs.Value.Name);
146 }
147 foreach (var cs in Fiji.Macro.Functions)
148 {
149 if (cs.Value[0].Name.StartsWith(pred))
150 preds.Add(cs.Value[0].Name);
151 }
152 predLabel.Text = "";
153 foreach(var item in preds)
154 {
155 if (this.AllocatedWidth > Measure(predLabel.Text))
156 {
157 predLabel.Text += item + ", ";
158 }
159 else break;
160 }
161 if(consoleBox.Buffer.Text.EndsWith('\t'))
162 {
163 string s = consoleBox.Buffer.Text.TrimEnd('\t');
164 skip = true;
165 consoleBox.Buffer.Text = s.Remove(s.Length - pred.Length, pred.Length) + preds[0];
166 }
167 }
