216 private void RoiView_RowActivated(object o, RowActivatedArgs args)
217 {
218 TreeIter iter;
219 if (roiView.Model.GetIter(out iter, args.Path))
220 {
221 string type = (string)roiView.Model.GetValue(iter, 0);
222 ROI.Type t;
223 if (Enum.TryParse<ROI.Type>(type,out t))
224 {
225 string id = (string)roiView.Model.GetValue(iter, 1);
226 string text = (string)roiView.Model.GetValue(iter, 2);
227 string bounds = (string)roiView.Model.GetValue(iter, 3);
228 if (rois.ContainsKey(id))
229 {
230 anno = rois[id];
231 if (App.viewer != null)
232 App.viewer.SetCoordinate(anno.coord.Z, anno.coord.C, anno.coord.T);
233
234 if (anno.type == ROI.Type.Line || anno.type == ROI.Type.Polygon ||
235 anno.type == ROI.Type.Polyline)
236 {
237 xBox.Sensitive = false;
238 yBox.Sensitive = false;
239 wBox.Sensitive = false;
240 hBox.Sensitive = false;
241 }
242 else
243 {
244 xBox.Sensitive = true;
245 yBox.Sensitive = true;
246 wBox.Sensitive = true;
247 hBox.Sensitive = true;
248 }
249 if (anno.type == ROI.Type.Rectangle || anno.type == ROI.Type.Ellipse)
250 {
251 pointBox.Sensitive = false;
252 pointXBox.Sensitive = false;
253 pointYBox.Sensitive = false;
254 }
255 else
256 {
257 pointBox.Sensitive = true;
258 pointXBox.Sensitive = true;
259 pointYBox.Sensitive = true;
260 }
261 xBox.Value = anno.X;
262 yBox.Value = anno.Y;
263 wBox.Value = anno.W;
264 hBox.Value = anno.H;
265 zBox.Value = anno.coord.Z;

267 tBox.Value = anno.coord.T;

338 public void UpdateAnnotationList()
339 {
340 rois.Clear();
341 Gtk.TreeStore store = new Gtk.TreeStore(typeof(string), typeof(string), typeof(string), typeof(string));
342 foreach (BioImage b in Images.images)
343 {
344 Gtk.TreeIter iter = store.AppendValues(System.IO.Path.GetFileName(b.Filename));
345 foreach (ROI r in b.Annotations)
346 {
347 rois.Add(r.type.ToString() + "," + r.id + "," + r.Text + "," + r.BoundingBox.ToString(),r);
348 store.AppendValues(iter, r.type.ToString(), r.id, r.Text, r.BoundingBox.ToString());
349 }
