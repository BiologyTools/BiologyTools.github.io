280 public static bool SetImageJPath()
281 {
282 string st = BioLib.Settings.GetSettings("ImageJPath");
283 if (st != "")
284 {
285 Fiji.ImageJPath = BioLib.Settings.GetSettings("ImageJPath");
286 return true;
287 }
288 string title = "Select ImageJ Executable Location";
289 if (OperatingSystem.IsMacOS())
290 title = "Select ImageJ Executable Location (Fiji.app/Contents/MacOS/ImageJ-macosx)";
291 Gtk.FileChooserDialog filechooser =
292 new Gtk.FileChooserDialog(title, Scripting.window,
293 FileChooserAction.Open,
294 "Cancel", ResponseType.Cancel,
295 "Save", ResponseType.Accept);
296 filechooser.SetCurrentFolder(System.IO.Path.GetDirectoryName(Environment.ProcessPath));
297 if (filechooser.Run() != (int)ResponseType.Accept)
298 return false;

305 private void Mi_ButtonPressEvent(object o, ButtonPressEventArgs args)
306 {
307 MenuItem m = (MenuItem)o;
308 if (m.Label.EndsWith(".ijm") || m.Label.EndsWith(".txt") && !m.Label.EndsWith(".cs"))

312 Fiji.RunOnImage(ImageView.SelectedImage,ma, BioConsole.headless, BioConsole.onTab, BioConsole.useBioformats, BioConsole.resultInNewTab);
313 }
314 else
315 Scripting.RunByName(m.Label);
316 bool con = false;
317 foreach (MenuItem item in recent.Children)
318 {
319 if (item.Label == m.Label)
320 con = true;
321 }
322 if (!con)
323 recent.Append(m);

461 private void TabsView_DragDataReceived(object o, DragDataReceivedArgs args)
462 {
463 // Convert the received data to a string
464 string receivedData = System.Text.Encoding.UTF8.GetString(args.SelectionData.Data);
465
466 // Parse the URIs (file paths)
467 string[] uris = receivedData.Split(new[] { '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
468 foreach (string uri in uris)
469 {
470 if (Uri.TryCreate(uri, UriKind.Absolute, out Uri fileUri) && fileUri.IsFile)
471 {
472 string filePath = fileUri.LocalPath;
473 AddTab(BioImage.OpenFile(filePath));
474 }
475 }
476
477 args.RetVal = true; // Indicate the drop was handled
478 }
