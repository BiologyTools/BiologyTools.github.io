151 static short[] ConvertByteArrayToShortArray(byte[] byteArray, bool littleEndian)
152 {
153 if (byteArray.Length % 2 != 0)
154 {
155 throw new ArgumentException("The length of the byte array must be even.");
156 }
157
158 // Each short is made from two bytes
159 short[] shortArray = new short[byteArray.Length / 2];
160
161 for (int i = 0; i < shortArray.Length; i++)
162 {
163 if (littleEndian)
164 {
165 // If the system is little-endian, reverse the byte order
166 shortArray[i] = (short)((byteArray[i * 2 + 1] << 8) | byteArray[i * 2]);
167 }
168 else
169 {
170 // If the system is big-endian, keep the byte order as is
171 shortArray[i] = (short)((byteArray[i * 2] << 8) | byteArray[i * 2 + 1]);
172 }
173 }
174 return shortArray;
175 }
176 public static ImagePlus GetImagePlus(BioImage bm)
177 {
178 ImageStack ims = new ImageStack(bm.SizeX, bm.SizeY);
179 if (bm.Buffers[0].PixelFormat == AForge.PixelFormat.Format16bppGrayScale)
180 {
181 // Populate the stack with 16-bit images
182 for (int t = 0; t < bm.SizeT; t++)
183 {
184 for (int c = 0; c < bm.SizeC; c++)
185 {
186 for (int z = 0; z < bm.SizeZ; z++)
187 {
188 short[] pixels = ConvertByteArrayToShortArray(bm.Buffers[bm.GetFrameIndex(z, c, t)].Bytes, true);
189 ImageProcessor ip = new ShortProcessor(bm.SizeX, bm.SizeY, pixels, java.awt.image.ColorModel.getRGBdefault());
190 ims.addSlice("Z" + z + "C" + c + "T" + t, ip);
191 }
192 }
193 }
194 }
195 else if (bm.Buffers[0].PixelFormat == AForge.PixelFormat.Format8bppIndexed)
196 {
197 // Create a grayscale IndexColorModel
198 byte[] grayscale = new byte[256];
199 for (int i = 0; i < 256; i++)
200 {
201 grayscale[i] = (byte)i;
202 }
203 IndexColorModel colorModel = new IndexColorModel(8, 256, grayscale, grayscale, grayscale);
204 for (int t = 0; t < bm.SizeT; t++)
205 {
206 for (int c = 0; c < bm.SizeC; c++)
207 {
208 for (int z = 0; z < bm.SizeZ; z++)
209 {
210 ImageProcessor ip = new ByteProcessor(bm.SizeX, bm.SizeY, bm.Buffers[bm.GetFrameIndex(z, c, t)].Bytes, colorModel);
211 ims.addSlice("Z" + z + "C" + c + "T" + t, ip);
212 }
213 }
214 }
215 }
216 else if (bm.Buffers[0].PixelFormat == AForge.PixelFormat.Format24bppRgb)
217 {
218 for (int t = 0; t < bm.SizeT; t++)
219 {
220 for (int c = 0; c < bm.SizeC; c++)
221 {
222 for (int z = 0; z < bm.SizeZ; z++)
223 {
224 byte[] bts = bm.Buffers[bm.GetFrameIndex(z, c, t)].Bytes;
225 // Convert the byte array to an int array for the RGB processor
226 int[] rgbPixels = new int[bm.SizeX * bm.SizeY];
227 for (int i = 0; i < rgbPixels.Length; i++)
228 {
229 int r = bts[i * 3] & 0xFF; // Red
230 int g = bts[i * 3 + 1] & 0xFF; // Green
231 int b = bts[i * 3 + 2] & 0xFF; // Blue
232 rgbPixels[i] = (r << 16) | (g << 8) | b; // Combine into RGB int
233 }
234 // Create a ColorProcessor
235 ImageProcessor ip = new ColorProcessor(bm.SizeX, bm.SizeY, rgbPixels);
236 ims.addSlice("Z" + z + "C" + c + "T" + t, ip);
237 }
238 }
239 }
240 }
241 else if (bm.Buffers[0].PixelFormat == AForge.PixelFormat.Format32bppArgb || bm.Buffers[0].PixelFormat == AForge.PixelFormat.Format32bppRgb)
242 {
243 for (int t = 0; t < bm.SizeT; t++)
244 {
245 for (int c = 0; c < bm.SizeC; c++)
246 {
247 for (int z = 0; z < bm.SizeZ; z++)
248 {
249 byte[] bts = bm.Buffers[bm.GetFrameIndex(z, c, t)].Bytes;
250 // Convert the byte array to an int array for the ColorProcessor
251 int[] rgbaPixels = new int[bm.SizeX * bm.SizeY];
252 for (int i = 0; i < rgbaPixels.Length; i++)
253 {
254 int r = bts[i * 4] & 0xFF; // Red
255 int g = bts[i * 4 + 1] & 0xFF; // Green
256 int b = bts[i * 4 + 2] & 0xFF; // Blue
257 int a = bts[i * 4 + 3] & 0xFF; // Alpha
258 rgbaPixels[i] = (a << 24) | (r << 16) | (g << 8) | b; // Combine into ARGB int
259 }
260 // Create a ColorProcessor with the ARGB int array
261 ImageProcessor ip = new ColorProcessor(bm.SizeX, bm.SizeY, rgbaPixels);
262 ims.addSlice("Z" + z + "C" + c + "T" + t, ip);
263 }
264 }
265 }
266 }
267 else if (bm.Buffers[0].PixelFormat == AForge.PixelFormat.Format48bppRgb)
268 {
269 bm.To16Bit();
270 // Populate the stack with 16-bit images
271 for (int t = 0; t < bm.SizeT; t++)
272 {
273 for (int c = 0; c < bm.SizeC; c++)
274 {
275 for (int z = 0; z < bm.SizeZ; z++)
276 {
277 short[] pixels = ConvertByteArrayToShortArray(bm.Buffers[bm.GetFrameIndex(z, c, t)].Bytes, true);
278 ImageProcessor ip = new ShortProcessor(bm.SizeX, bm.SizeY, pixels, java.awt.image.ColorModel.getRGBdefault());
279 ims.addSlice("Z" + z + "C" + c + "T" + t, ip);
280 }
281 }
282 }
283 }
284 ImagePlus imp = new ImagePlus(bm.ID, ims);
285 imp.setDimensions(bm.SizeC, bm.SizeZ, bm.SizeT);
286 return imp;

310 public static BioImage GetBioImage(ImagePlus pl, VolumeD vol, double PhysicalX, double PhysicalY, double PhysicalZ)
311 {
312 BioImage bm = new BioImage(pl.getTitle());
313 ImageStack st = pl.getImageStack();
314 int b = pl.getBitDepth();
315 int slices = pl.getNSlices();
316 int chs = pl.getNChannels();
317 int frs = pl.getNFrames();
318 int rgb = (int)(pl.getBytesPerPixel() * 8);
319 bool isrgb = isRGB(pl);
320 bm.UpdateCoords(slices, chs, frs, BioImage.Order.TCZ);
321 for (int t = 0; t < frs; t++)
322 {
323 for (int c = 0; c < chs; c++)
324 {
325 for (int z = 0; z < slices; z++)
326 {
327 AForge.Bitmap bmp;
328 if (!isrgb)
329 {
330 if (rgb > 8)
331 {
332 bmp = new AForge.Bitmap(st.getWidth(), st.getHeight(), AForge.PixelFormat.Format16bppGrayScale);
333 bm.Buffers.Add(bmp);
334 }
335 else
336 {
337 bmp = new AForge.Bitmap(st.getWidth(), st.getHeight(), AForge.PixelFormat.Format8bppIndexed);
338 bm.Buffers.Add(bmp);
339 }
340 }
341 else
342 {
343 bmp = new AForge.Bitmap(st.getWidth(), st.getHeight(), AForge.PixelFormat.Format24bppRgb);
344 bm.Buffers.Add(bmp);
345 }
346 int ind = bm.GetFrameIndex(z, c, t);
347 if (!isrgb)
348 {
349 for (int y = 0; y < pl.getHeight(); y++)
350 {
351 for (int x = 0; x < pl.getWidth(); x++)
352 {
353 double d = st.getVoxel(x, y, ind);
354 if (bm.Buffers[0].PixelFormat == PixelFormat.Format16bppGrayScale)
355 bmp.SetValue(x, y, (ushort)d);
356 else if (bm.Buffers[0].PixelFormat == PixelFormat.Format8bppIndexed)
357 bmp.SetValue(x, y, (byte)d);
358 }
359 }
360 }
361 else
362 {
363 int[] rgbPixels = getRGBPixelsFromSlice(st, ind);
364 // Get the ImageProcessor for the specified slice
365 ImageProcessor ip = st.getProcessor(ind + 1); // ImageStack uses 1-based indexing
366 // Cast to ColorProcessor to access RGB data
367 ColorProcessor colorProcessor = (ColorProcessor)ip;
368 for (int y = 0; y < pl.getHeight(); y++)
369 {
370 for (int x = 0; x < pl.getWidth(); x++)
371 {
372 int v = colorProcessor.getPixel(x, y);
373 int rv = (v >> 16) & 0xFF; // Red
374 int gv = (v >> 8) & 0xFF; // Green
375 int bv = v & 0xFF; // Blue
376 bmp.SetValue(x, y, 0, (byte)rv);
377 bmp.SetValue(x, y, 1, (byte)gv);
378 bmp.SetValue(x, y, 2, (byte)bv);
379 }
380 }
381
382 }
383 }
384 }
385 }
386 for (int t = 0; t < chs; t++)
387 {
388 if (isrgb)
389 bm.Channels.Add(new AForge.Channel(t, rgb, 3));
390 else
391 bm.Channels.Add(new AForge.Channel(t, rgb, 1));
392 if (t == 0)
393 {
394 bm.rgbChannels[0] = 0;
395 }
396 else
397 if (t == 1)
398 {
399 bm.rgbChannels[1] = 1;
400 }
401 else
402 if (t == 2)
403 {
404 bm.rgbChannels[2] = 2;
405 }
406 }
407 bm.Resolutions.Add(new Resolution(pl.getWidth(), pl.getHeight(), bm.Buffers[0].PixelFormat, PhysicalX, PhysicalY, PhysicalZ, vol.Location.X, vol.Location.Y, vol.Location.X));
408 bm.Volume = vol;
409 bm.littleEndian = BitConverter.IsLittleEndian;
410 bm.seriesCount = 1;
411 bm.bitsPerPixel = b;
412 BioImage.AutoThreshold(bm, true);
413 if (rgb > 8)
414 bm.StackThreshold(true);
415 else
416 bm.StackThreshold(false);
417
418 if (bm.RGBChannelCount == 4)
419 {
420 bm.Channels.Last().SamplesPerPixel = 4;
421 }
422 return bm;
423 }
