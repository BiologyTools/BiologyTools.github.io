105 {
106 foreach (var dropDownItem in item.DropDownItems)
107 {
108 if (dropDownItem.GetType() == typeof(ToolStripMenuItem))
109 {
110 if (((ToolStripMenuItem)dropDownItem).HasDropDownItems)
111 {
112 foreach (ToolStripMenuItem subItem in GetItems((ToolStripMenuItem)dropDownItem))
113 yield return subItem;

115 yield return (ToolStripMenuItem)dropDownItem;

152 public static ToolStripItem GetMenuItemFromPath(string s, Function f)
153 {
154 if (s == "" || s == null)
155 return null;
156 string[] sts = s.Split('/');
157 start:
158 List<ToolStripMenuItem> allItems = GetMenuItems();
159 //Find path or create it.
160 bool found = false;
161 ToolStripMenuItem item = null;
162
163 for (int t = 0; t < sts.Length; t++)
164 {
165 found = false;
166 for (int i = 0; i < allItems.Count; i++)
167 {
168 if (allItems[i].Text == sts[t])
169 {
170 item = allItems[i];
171 found = true;
172 if (t == sts.Length - 1)
173 return allItems[i];
174 }
175 }
176 if (!found)
177 {
178 if(t == 0)
179 {
180 tabsView.MainMenuStrip.Items.Add(sts[t]);
181 goto start;
182 }
183 else if(t > 0 && t < sts.Length)
184 {
185 ToolStripMenuItem itm = new ToolStripMenuItem();
186 itm.Tag = f;
187 itm.Name = f.Name;
188 item.DropDownItems.Add(f.Name, null, ItemClicked);
189 return item;
190 }
191 else
192 {
193 item.DropDownItems.Add(sts[t]);
194 }
195 }
196 }
197 return item;
198 }

403 private static void ItemClicked(object sender, EventArgs e)
404 {
405 ToolStripMenuItem ts = (ToolStripMenuItem)sender;
406 if(ts.Text.EndsWith(".dll"))
407 {
408 try
409 {
410 Plugin.Plugins[ts.Text].Execute(new string[] { });
411 }
412 catch (Exception ex)
413 {
414 Console.WriteLine(ex.ToString());
415 }
416
417 }
418 else if(ts.Text.EndsWith(".ijm"))
419 {
420 string st = File.ReadAllText(Path.GetDirectoryName(ImageJ.ImageJPath) + "/macros/" + ts.Text);
421 ImageJ.RunOnImage(st, BioConsole.headless, BioConsole.onTab, BioConsole.useBioformats, BioConsole.newTab);
422 }
423 else
424 {
425 Function f = (Function)ts.Tag;
426 f.PerformFunction(true);
427 }
