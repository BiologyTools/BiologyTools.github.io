201 private void UpdateValues()
202 {
203 if (channelsBox.Active == -1)
204 return;
205 sampleBox.Adjustment.Upper = SelectedChannel.range.Length - 1;
206 if (minBox.Adjustment.Upper < SelectedChannel.range[(int)sampleBox.Value].Min || maxBox.Adjustment.Upper < SelectedChannel.range[(int)sampleBox.Value].Max)
207 {
208 minBox.Value = 0;
209 maxBox.Value = ushort.MaxValue;
210 }
211 else
212 {
213 minBox.Value = Channels[channelsBox.Active].range[(int)sampleBox.Value].Min;
214 maxBox.Value = Channels[channelsBox.Active].range[(int)sampleBox.Value].Max;
215 }
216 if (hist != null)
217 {
218 //hist.Statistics = Channels[channelsBox.Active].statistics;
219 hist.UpdateChannel(SelectedChannel);
220 hist.UpdateView();
221 }
222 float max = float.MinValue;
223 float min = float.MaxValue;
224 foreach (Channel c in Channels)
225 {
226 for (int i = 0; i < c.SamplesPerPixel; i++)
227 {
228 if (max < c.range[i].Max)
229 max = c.range[i].Max;
230 if (min > c.range[i].Min)
231 min = c.range[i].Min;
232 }
233 }
234 graphMax.Value = max;
235 graphMinBox.Value = min;
236 fluorBox.Text = SelectedChannel.Fluor;
237 emissionBox.Value = SelectedChannel.Emission;
238 excitationBox.Value = SelectedChannel.Excitation;
239 }
