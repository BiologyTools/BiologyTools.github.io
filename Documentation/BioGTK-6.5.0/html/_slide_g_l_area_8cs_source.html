67 private const string VertexShaderSource = @"
68#version 330 core
69
70layout(location=0) in vec2 aPos;
71layout(location=1) in vec2 aUV;
72
73uniform vec2 pos; // pixel-space top-left of tile
74uniform vec2 size; // pixel-space size of tile
75uniform vec2 viewportSize; // (width, height) in pixels
76
77out vec2 uv;
78
79void main()
80{
81 // aPos is in [0,1] x [0,1]
82 vec2 pixelPos = aPos * size + pos;
83
84 // Convert from pixel coords to NDC (-1..1), with Y flipped for GL
85 vec2 ndc;
86 ndc.x = (pixelPos.x / viewportSize.x) * 2.0 - 1.0;
87 ndc.y = 1.0 - (pixelPos.y / viewportSize.y) * 2.0;
88
89 gl_Position = vec4(ndc, 0.0, 1.0);
90 uv = aUV;
91}
92";
93
94 private const string FragmentShaderSource = @"
95#version 330 core
96
97in vec2 uv;
98out vec4 FragColor;
99uniform sampler2D tex;
100
101void main()
102{
103 // Fix: Removed '1.0 - uv.y'.
104 // Image data is usually Top-Down. OpenGL Textures are Bottom-Up.
105 // This implicit flip means UV(0,0) (Top-Left Quad) maps to Texture Bottom (Image Top).
106 // This is the correct orientation for Top-Left origin rendering.
107 FragColor = texture(tex, vec2(uv.x, uv.y));
108}
109";

163 private void InitializeGL()
164 {
165 // Compile shaders
166 int vertexShader = CompileShader(ShaderType.VertexShader, VertexShaderSource);
167 int fragmentShader = CompileShader(ShaderType.FragmentShader, FragmentShaderSource);
168
169 _shaderProgram = GL.CreateProgram();
170 GL.AttachShader(_shaderProgram, vertexShader);
171 GL.AttachShader(_shaderProgram, fragmentShader);
172 GL.LinkProgram(_shaderProgram);
173
174 GL.GetProgram(_shaderProgram, GetProgramParameterName.LinkStatus, out int linkStatus);
175 if (linkStatus == 0)
176 {
177 string infoLog = GL.GetProgramInfoLog(_shaderProgram);
178 throw new Exception($"Shader link failed: {infoLog}");
179 }
180
181 GL.DeleteShader(vertexShader);
182 GL.DeleteShader(fragmentShader);
183
184 // Cache uniform locations
185 _locPos = GL.GetUniformLocation(_shaderProgram, "pos");
186 _locSize = GL.GetUniformLocation(_shaderProgram, "size");
187 _locViewportSize = GL.GetUniformLocation(_shaderProgram, "viewportSize");
188 _locTex = GL.GetUniformLocation(_shaderProgram, "tex");
189
190 // Create quad VAO/VBO for tile rendering
191 float[] quadVertices =
192 {
193 // pos uv
194 0, 0, 0, 0,
195 1, 0, 1, 0,
196 1, 1, 1, 1,
197
198 0, 0, 0, 0,
199 1, 1, 1, 1,
200 0, 1, 0, 1
201 };
202
203 _vao = GL.GenVertexArray();
204 _vbo = GL.GenBuffer();
205
206 GL.BindVertexArray(_vao);
207 GL.BindBuffer(BufferTarget.ArrayBuffer, _vbo);
208 GL.BufferData(BufferTarget.ArrayBuffer, quadVertices.Length * sizeof(float),
209 quadVertices, BufferUsageHint.StaticDraw);
210
211 // Position attribute
212 GL.VertexAttribPointer(0, 2, VertexAttribPointerType.Float, false, 4 * sizeof(float), 0);
213 GL.EnableVertexAttribArray(0);
214
215 // UV attribute
216 GL.VertexAttribPointer(1, 2, VertexAttribPointerType.Float, false, 4 * sizeof(float), 2 * sizeof(float));
217 GL.EnableVertexAttribArray(1);
218
219 GL.BindVertexArray(0);
220 }

319 private void RenderTiles(int width, int height)
320 {
321 // Multiply by ScaleFactor to handle High-DPI screens correctly
322 int scale = ScaleFactor;
323 GL.Viewport(0, 0, width * scale, height * scale);
324 GL.ClearColor(0.2f, 0.2f, 0.2f, 1f);
325 GL.Clear(ClearBufferMask.ColorBufferBit);
326
327 if (TilesToRender.Count == 0)
328 {
329 Console.WriteLine("WARNING: No tiles to render!");
330 return;
331 }
332
333 Console.WriteLine($"Rendering {TilesToRender.Count} tiles at viewport {width}x{height}");
334
335 GL.Enable(EnableCap.Blend);
336 GL.BlendFunc(BlendingFactor.SrcAlpha, BlendingFactor.OneMinusSrcAlpha);
337
338 GL.UseProgram(_shaderProgram);
339 GL.BindVertexArray(_vao);
340
341 // Set viewport size uniform (constant for all tiles this frame)
342 GL.Uniform2(_locViewportSize, (float)width, (float)height);
343 GL.Uniform1(_locTex, 0);
344
345 GL.ActiveTexture(TextureUnit.Texture0);
346
347 int renderedCount = 0;
348 foreach (var tile in TilesToRender)
349 {
350 if (!_textureCache.TryGetValue(tile.Index, out int texId))
351 {
352 Console.WriteLine($"WARNING: Tile {tile.Index} not in texture cache!");
353 continue;
354 }
355
356 GL.Uniform2(_locPos, tile.ScreenX, tile.ScreenY);
357 GL.Uniform2(_locSize, tile.ScreenWidth, tile.ScreenHeight);
358
359 Console.WriteLine($"Rendering tile {tile.Index}: pos=({tile.ScreenX}, {tile.ScreenY}), size=({tile.ScreenWidth}, {tile.ScreenHeight})");
360
361 GL.BindTexture(TextureTarget.Texture2D, texId);
362 GL.DrawArrays(PrimitiveType.Triangles, 0, 6);
363 renderedCount++;
364 }
365
366 Console.WriteLine($"Successfully rendered {renderedCount} tiles");
367
368 GL.BindVertexArray(0);
369 GL.UseProgram(0);
370 GL.Disable(EnableCap.Blend);
371 }

390 public void UploadTileTexture(TileIndex index, byte[] pixelData, int tileWidth, int tileHeight)
391 {
392 if (!_glInitialized)
393 {
394 Console.WriteLine("WARNING: GL not initialized, cannot upload texture");
395 return;
396 }
397
398 if (pixelData == null || pixelData.Length == 0)
399 {
400 Console.WriteLine($"WARNING: Empty pixel data for tile {index}");
401 return;
402 }
403
404 MakeCurrent();
405
406 // Check if already cached
407 if (_textureCache.ContainsKey(index))
408 {
409 Console.WriteLine($"Tile {index} already cached");
410 return;
411 }
412
413 // Evict old textures if cache is full
414 if (_textureCache.Count >= MAX_CACHED_TEXTURES)
415 {
416 EvictOldestTextures(MAX_CACHED_TEXTURES / 4);
417 }
418
419 // Create and upload texture
420 int tex = GL.GenTexture();
421 GL.BindTexture(TextureTarget.Texture2D, tex);
422
423 GL.PixelStore(PixelStoreParameter.UnpackAlignment, 1);
424
425 // CRITICAL FIX: Use Rgba instead of Bgra for internal format consistency
426 GL.TexImage2D(TextureTarget.Texture2D, 0, PixelInternalFormat.Rgba,
427 tileWidth, tileHeight, 0, PixelFormat.Rgba, PixelType.UnsignedByte, pixelData);
428
429 GL.TexParameter(TextureTarget.Texture2D, TextureParameterName.TextureMinFilter, (int)TextureMinFilter.Linear);
430 GL.TexParameter(TextureTarget.Texture2D, TextureParameterName.TextureMagFilter, (int)TextureMagFilter.Linear);
431 GL.TexParameter(TextureTarget.Texture2D, TextureParameterName.TextureWrapS, (int)TextureWrapMode.ClampToEdge);
432 GL.TexParameter(TextureTarget.Texture2D, TextureParameterName.TextureWrapT, (int)TextureWrapMode.ClampToEdge);
433
434 GL.BindTexture(TextureTarget.Texture2D, 0);
435
436 _textureCache[index] = tex;
437 Console.WriteLine($"Uploaded tile {index} as texture {tex} ({tileWidth}x{tileHeight}, {pixelData.Length} bytes)");
438 }
