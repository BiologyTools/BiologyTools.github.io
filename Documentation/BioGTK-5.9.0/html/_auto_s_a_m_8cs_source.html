93 double[,] BuildPointGrid(int n_per_side)
94 {
95 double offset = 1.0 / (2 * n_per_side);
96 double[] points_one_side = Enumerable.Range(0, n_per_side)
97 .Select(i => offset + i * (1.0 - 2 * offset) / (n_per_side - 1))
98 .ToArray();
99
100 double[,] points = new double[n_per_side * n_per_side, 2];
101
102 for (int i = 0; i < n_per_side; i++)
103 {
104 for (int j = 0; j < n_per_side; j++)
105 {
106 int index = i * n_per_side + j;
107 points[index, 0] = points_one_side[i];
108 points[index, 1] = points_one_side[j];
109 }
110 }
111
112 return points;
113 }

120 void generateCropBoxes(int orgWid, int orgHei, int n_layers, float overlap_ratio,
121 ref List<List<int>> crop_boxes, ref List<int> layer_idxs)
122 {
123 int im_h = orgHei;
124 int im_w = orgWid;
125
126 int short_side = Math.Min(im_h, im_w);
127 //Original image
128 crop_boxes.Add(new List<int> { 0, 0, im_w, im_h });
129 layer_idxs.Add(0);
130
131 for (int i_layer = 0; i_layer < n_layers; i_layer++)
132 {
133 int n_crops_per_side = (int)Math.Pow(2, i_layer + 1);
134 int overlap = (int)(overlap_ratio * short_side * (2 / n_crops_per_side));
135
136 int crop_w = crop_len(im_w, n_crops_per_side, overlap);
137 int crop_h = crop_len(im_h, n_crops_per_side, overlap);
138
139 List<int> crop_box_x0 = new List<int>();
140 List<int> crop_box_y0 = new List<int>();
141
142 for (int i = 0; i < n_crops_per_side; i++)
143 {
144 crop_box_x0.Add((crop_w - overlap) * i);
145 crop_box_y0.Add((crop_h - overlap) * i);
146 }
147
148 foreach (int x0 in crop_box_x0)
149 {
150 foreach (int y0 in crop_box_y0)
151 {
152 List<int> box = new List<int>
153 {
154 x0,
155 y0,
156 Math.Min(x0 + crop_w, im_w),
157 Math.Min(y0 + crop_h, im_h)
158 };
159
160 crop_boxes.Add(box);
161 layer_idxs.Add(i_layer + 1);
162 }
163 }
164
165 }

168 {
169 return (int)(Math.Ceiling((double)(overlap * (n_crops - 1) + orig_len) / n_crops));

171 IEnumerable<List<object>> BatchIterator(int batch_size, params object[] args)
172 {
173
174 int n_batches = ((Array)args[0]).Length / batch_size + (((Array)args[0]).Length % batch_size != 0 ? 1 : 0);
175 for (int b = 0; b < n_batches; b++)
176 {
177 List<object> batch = new List<object>();
178 foreach (object arg in args)
179 {
180 Array arr = (Array)arg;
181 int start = b * batch_size;
182 int end = (b + 1) * batch_size;
183 if (end > arr.Length)
184 end = arr.Length;
185
186 Array slice = Array.CreateInstance(arr.GetType().GetElementType(), end - start);
187 Array.Copy(arr, start, slice, 0, end - start);
188 batch.Add(slice);
189 }
190
191 yield return batch;
192 }
