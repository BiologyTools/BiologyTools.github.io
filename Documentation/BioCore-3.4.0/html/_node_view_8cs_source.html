127 {
128 if (Images.images.Count != treeView.Nodes[0].Nodes.Count)
129 {
130 //If image count is not same as node count we refresh whole tree.
131 InitNodes();

133 }

137 //TreeNode node = new TreeNode();

140
141 TreeNode rois = node.node.Nodes[1];
142 if (im.Annotations.Count != rois.Nodes.Count)
143 {
144 //If ROI count is not same as node count we refresh annotations.
145 rois.Nodes.Clear();
146 foreach (ROI an in im.Annotations)
147 {
148 Node roi = new Node(an, Node.DataType.roi);
149 rois.Nodes.Add(roi.node);
150 }
151 }
152 else
153 for (int i = 0; i < im.Annotations.Count; i++)
154 {
155 TreeNode roi = rois.Nodes[i];
