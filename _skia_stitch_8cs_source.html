189 public async Task<SKImage> StitchViewportAsync(
190 PointD origin,
191 double resolution,
192 ZCT coordinate,
193 int viewportWidth = 600,
194 int viewportHeight = 400)
195 {
196 // Calculate viewport parameters
197 int level = CalculateLevelFromResolution(resolution);
198 var tileRequests = CalculateTileRequests(origin, viewportWidth, viewportHeight, level, coordinate, resolution);
199 if (viewportWidth <= 1 || viewportHeight <= 1)
200 {
201 viewportWidth = 600;
202 viewportHeight = 400;
203 }
204 // Create a NEW surface for this render - no shared state, fully parallel-safe
205 using (var surface = SKSurface.Create(new SKImageInfo(
206 viewportWidth,
207 viewportHeight,
208 SKColorType.Rgba8888,
209 SKAlphaType.Premul)))
210 {
211 if (surface == null)
212 {
213 Console.WriteLine("Failed to create SKSurface");
214 return null;
215 }
216
217 var canvas = surface.Canvas;
218 canvas.Clear(SKColors.Gray);
219
220 // Fetch and composite all tiles
221 await CompositeTilesAsync(canvas, tileRequests, origin, resolution);
222
223 // Create and return snapshot
224 return surface.Snapshot();
225 } // Surface automatically disposed here

267 private async Task<SKImage> FetchTileAsync(TileRequest request, ZCT coordinate)
268 {
269 string key = request.GetCacheKey();
270
271 // 1. Check cache again to be sure
272 if (_tileCache.TryGetValue(key, out var cached)) return cached.Image;
273
274 // 2. Check if already being fetched
275 if (_pendingTileFetches.TryGetValue(key, out var pendingTask)) return await pendingTask;
276
277 // 3. Create the fetch task
278 var fetchTask = Task.Run(async () =>
279 {
280 try
281 {
282 byte[] data;
283 if (_isOpenSlide)
284 {
285 TileInfo tf = new TileInfo { Index = request.Index, Extent = request.Extent };
286 // OpenSlide usually takes Level, X-index, Y-index
287 data = _openSlideSource.GetTile(tf);
288 }
289 else
290 {
291 // SlideBase usually takes the Index object and ZCT
292 TileInfo tf = new TileInfo { Index = request.Index, Extent = request.Extent };
293 // OpenSlide usually takes Level, X-index, Y-index
294 data = _slideSource.GetTile(tf,request.Coordinate);
295 }
296
297 if (data == null || data.Length == 0) return null;
298
299 // Convert to Skia Image (using the shear-fix logic from before)
300 var levelResolution = CalculateResolutionFromLevel(request.Index.Level);
301 var image = ConvertTileDataToSKImage(data, request);
302
303 if (image != null)
304 {
305 var cachedTile = new CachedTile(image, DateTime.Now.ToString());
306 _tileCache.AddOrUpdate(key, cachedTile, (k, old) => cachedTile);
307 return image;
308 }
309 }
310 catch (Exception ex)
311 {
312 Console.WriteLine($"Fetch error for tile {key}: {ex.Message} + {ex.StackTrace}");
313 }
314 finally
315 {
316 _pendingTileFetches.TryRemove(key, out _);
317 }
318 return null;
319 });
320
321 _pendingTileFetches[key] = fetchTask;
322 return await fetchTask;
323 }

328 private SKImage FetchTileSync(TileRequest request)
329 {
330 string cacheKey = request.GetCacheKey();
331
332 // Check cache
333 if (_tileCache.TryGetValue(cacheKey, out var cachedTile))
334 {
335 cachedTile.LastAccessed = DateTime.UtcNow;
336 return cachedTile.Image;
337 }
338
339 // Fetch directly
340 try
341 {
342 byte[] tileData = FetchTileDataSync(request);
343 if (tileData == null || tileData.Length == 0)
344 return null;
345
346 SKImage skImage = ConvertTileDataToSKImage(tileData, request);
347
348 if (skImage != null)
349 {
350 AddToCache(cacheKey, skImage);
351 }
352
353 return skImage;
354 }
355 catch (Exception ex)
356 {
357 Console.WriteLine($"Error fetching tile sync {cacheKey}: {ex.Message}");
358 return null;
359 }

462 private async Task CompositeTilesAsync(
463 SKCanvas canvas,
464 List<TileRequest> tileRequests,
465 PointD origin,
466 double resolution)
467 {
468 if (tileRequests.Count == 0) return;
469
470 // Fetch all tiles in parallel
471 var tileTasks = tileRequests.Select(async req =>
472 {
473 var image = await FetchTileAsync(req, req.Coordinate);
474 return new { Request = req, Image = image };
475 }).ToArray();
476
477 var results = await Task.WhenAll(tileTasks);
478
479 using (var paint = new SKPaint { IsAntialias = false })
480 {
481 foreach (var result in results)
482 {
483 if (result.Image == null) continue;
484 DrawTileToCanvas(canvas, result.Request, origin, resolution, result.Image, paint);
485 }
486 }

491 private void DrawTileToCanvas(
492 SKCanvas canvas,
493 TileRequest request,
494 PointD viewportOrigin,
495 double resolution,
496 SKImage tileImage,
497 SKPaint paint = null)
498 {
499 if (tileImage == null) return;
500
501 // X: (Tile Left - Viewport Left) / Resolution
502 float destX = (float)((request.Extent.MinX - viewportOrigin.X) / resolution);
503
504 // Y: (Viewport Top - Tile Top) / Resolution
505 // In BruTile, MaxY is the Top of the tile
506 float destY = (float)((viewportOrigin.Y - request.Extent.MaxY) / resolution);
507
508 // Calculate dimensions based on the Extent to ensure precision
509 float destW = (float)(request.Extent.Width / resolution);
510 float destH = (float)(request.Extent.Height / resolution);
511
512 var destRect = new SKRect(destX, destY, destX + destW, destY + destH);
513
514 // Use Sampling if provided in the class, otherwise use the paint object
515 if (paint != null)
516 canvas.DrawImage(tileImage, destRect, paint);
517 else
518 canvas.DrawImage(tileImage, destRect, Sampling);
519 }

550 private List<TileRequest> CalculateTileRequests(PointD origin, int viewportWidth, int viewportHeight, int level, ZCT coordinate, double resolution)
551 {
552 // origin.X/Y is the Top-Left of the viewport in World Space.
553 // In BruTile (OSM/TMS style), Y increases UPWARDS.
554
555 double minX = origin.X;
556 double maxX = origin.X + (viewportWidth * resolution);
557
558 // Because Y increases UP, the Top of the screen (origin.Y)
559 // is the MAXIMUM Y value.
560 double maxY = origin.Y;
561
562 // The Bottom of the screen is the MINIMUM Y value.
563 double minY = origin.Y - (viewportHeight * resolution);
564
565 // Create the extent (MinX, MinY, MaxX, MaxY)
566 var viewportExtent = new Extent(minX, minY, maxX, maxY);
567
568 IEnumerable<TileInfo> tileInfos;
569 if (_isOpenSlide)
570 {
571 tileInfos = _openSlideSource.Schema.GetTileInfos(viewportExtent, level);
572 }
573 else
574 {
575 tileInfos = _slideSource.Schema.GetTileInfos(viewportExtent, level);
576 }
577
578 var requests = new List<TileRequest>();
579 foreach (var info in tileInfos)
580 {
581 requests.Add(new TileRequest(info, coordinate));
582 }
583 return requests;
584 }

592 private void AddToCache(string key, SKImage image)
593 {
594 var cachedTile = new CachedTile(image, key);
595
596 lock (_cacheLock)
597 {
598 _tileCache[key] = cachedTile;
599
600 // Check if we need to evict
601 long currentCacheSize = CalculateTotalCacheSize();
602
603 if (currentCacheSize > _maxCacheSize)
604 {
605 EvictLeastRecentlyUsedTiles(currentCacheSize - _maxCacheSize);
606 }
607 }
608 }
609
613 private void EvictLeastRecentlyUsedTiles(long bytesToFree)
614 {
615 var sortedTiles = new List<CachedTile>(_tileCache.Values);
616 sortedTiles.Sort((a, b) => a.LastAccessed.CompareTo(b.LastAccessed));
617
618 long freedBytes = 0;
619 foreach (var tile in sortedTiles)
620 {
621 if (freedBytes >= bytesToFree)
622 break;
623
624 if (_tileCache.TryRemove(tile.Key, out var removed))
625 {
626 freedBytes += removed.SizeBytes;
627 removed.Image?.Dispose();
628 }
629 }
630 }
631
635 private long CalculateTotalCacheSize()
636 {
637 long total = 0;
638 foreach (var tile in _tileCache.Values)
639 {
640 total += tile.SizeBytes;
641 }
642 return total;
643 }
644
648 public void ClearCache()
649 {
650 lock (_cacheLock)
651 {
652 foreach (var tile in _tileCache.Values)
653 {
654 tile.Image?.Dispose();
655 }
656 _tileCache.Clear();
657 }

668 private SKImage ConvertTileDataToSKImage(byte[] tileData, TileRequest request)
669 {
670 try
671 {
672 // 1. Try standard decoding (JPEG/PNG/etc)
673 using (var data = SKData.CreateCopy(tileData))
674 {
675 var image = SKImage.FromEncodedData(data);
676 if (image != null) return image;
677 }
678
679 // 2. Fallback for Raw Pixels (The "Shear" Fix)
680 // Use the Schema's defined TileWidth/Height, NOT the calculated resolution width
681 int width = _isOpenSlide ? _openSlideSource.Schema.GetTileWidth(request.Index.Level)
682 : _slideSource.Schema.GetTileWidth(request.Index.Level);
683 int height = _isOpenSlide ? _openSlideSource.Schema.GetTileHeight(request.Index.Level)
684 : _slideSource.Schema.GetTileHeight(request.Index.Level);
685
686 var info = new SKImageInfo(width, height, SKColorType.Bgra8888, SKAlphaType.Premul);
687 // If decode fails, create from raw RGBA data
688 // CRITICAL: Don't dispose SKData until image is disposed
689 var imageInfo = new SKImageInfo(width, height, SKColorType.Rgba8888, SKAlphaType.Premul);
690 var rawData = SKData.CreateCopy(tileData);
691 var pixmap = new SKPixmap(imageInfo, rawData.Data, imageInfo.RowBytes);
692
693 // Use release delegate to properly manage lifetime
694 return SKImage.FromPixels(pixmap, (address, context) =>
695 {
696 rawData?.Dispose();
697 }, null);
698 }
699 catch (Exception ex)
700 {
701 Console.WriteLine($"Error converting tile data: {ex.Message}");
702 return null;
703 }
704 }
