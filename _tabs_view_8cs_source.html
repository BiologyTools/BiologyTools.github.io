148 public static TabsView Create()
149 {
150 Builder builder = new Builder(new FileStream(System.IO.Path.GetDirectoryName(Environment.ProcessPath) + "/" + "Glade/TabsView.glade", FileMode.Open));
151 return new TabsView(builder, builder.GetObject("TabsView").Handle);

154 /* Setting up the UI. */
155 protected TabsView(Builder builder, IntPtr handle) : base(handle)
156 {
157 _builder = builder;
158 App.tabsView = this;
159 builder.Autoconnect(this);
160 filteredMenu.Active = true;
161 SetupHandlers();
162 Function.InitializeMainMenu();
163 filechooser =
164 new Gtk.FileChooserDialog("Choose file to open",
165 this,
166 FileChooserAction.Open,
167 "Cancel", ResponseType.Cancel,
168 "OK", ResponseType.Accept);
169 filechooser.SelectMultiple = true;
170 Menu m = new Menu();
171 foreach (RotateFlipType flip in (RotateFlipType[])Enum.GetValues(typeof(RotateFlipType)))
172 {
173 MenuItem mi = new MenuItem(flip.ToString());
174 mi.ButtonPressEvent += RotateFlip_ButtonPressEvent;
175 m.Append(mi);
176 }
177 rotateFlipMenu.Submenu = m;
178 rotateFlipMenu.ShowAll();
179 }

351 private void ExportROIsFromImageJMenu_ButtonPressEvent(object o, ButtonPressEventArgs args)
352 {
353 Gtk.FileChooserDialog filechooser =
354 new Gtk.FileChooserDialog("Set ROI filename to save",
355 this,
356 FileChooserAction.Save,
357 "Cancel", ResponseType.Cancel,
358 "Save", ResponseType.Accept);
359 if (filechooser.Run() != (int)ResponseType.Accept)
360 return;
361 int i = 1;
362 foreach (ROI item in ImageView.SelectedImage.Annotations)
363 {
364 string f = filechooser.Filename;
365 if (!f.EndsWith(".roi"))
366 f += i + ".roi";
367 else
368 {
369 f = f.Replace(".roi","") + i + ".roi";
370 }
371 ImageJ.RoiEncoder.save(item, f);
372 i++;
373 }
374 filechooser.Hide();
375 }

409 private void TabsView_WindowStateEvent(object o, WindowStateEventArgs args)
410 {
411 if ((args.Event.ChangedMask & Gdk.WindowState.Iconified) != 0)
412 {
413 foreach (ImageView v in viewers)
414 {
415 v.Hide();
416 }
417 }
418
419 else if ((args.Event.ChangedMask & Gdk.WindowState.Maximized) == 0 && (args.Event.ChangedMask & Gdk.WindowState.Iconified) == 0)
420 {
421 foreach (ImageView v in viewers)
422 {
423 if(!v.Visible)
424 v.Present();
425 }
426 }
