245 private static void RunScript()
246 {
247 Script rn = scripts[scriptName];
248 rn.ex = null;
249 if (rn.type == ScriptType.imagej)
250 {
251 try
252 {
253 rn.done = false;
254 ImageJ.RunString(rn.scriptString,"", false);
255 rn.done = true;
256 }
257 catch (Exception e)
258 {
259 rn.ex = e;
260 }
261 }
262 else
263 {
264 try
265 {
266 rn.done = false;
267 rn.script = CSScript.Evaluator.LoadCode(rn.scriptString);
268 rn.obj = rn.script.Load();
269 rn.output = rn.obj.ToString();
270 rn.done = true;
271
272 }
273 catch (Exception e)
274 {
275 rn.ex = e;
276 error = e.Message + Environment.NewLine + e.StackTrace;
277 }
278 }
