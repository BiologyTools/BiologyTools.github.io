364 private bool NeedsUpdate(
365 PointD origin,
366 int width,
367 int height,
368 double resolution,
369 ZCT coordinate)
370 {
371 const double EPSILON = 0.0001;
372
373 return Math.Abs(origin.X - _lastOrigin.X) > EPSILON ||
374 Math.Abs(origin.Y - _lastOrigin.Y) > EPSILON ||
375 Math.Abs(resolution - _lastResolution) > EPSILON ||
376 width != _lastWidth ||
377 height != _lastHeight ||
378 coordinate.Z != _lastCoordinate.Z ||
379 coordinate.C != _lastCoordinate.C ||
380 coordinate.T != _lastCoordinate.T;
381 }
